Quick usage

<!-- 1. Import the stylesheet -->
<link rel="stylesheet" href="solar-eclipse.css" />

<!-- 2. Structure — requires corona & occluder -->
<div class="solar-eclipse" role="status" aria-label="Loading">
  <div class="solar-eclipse__corona"></div>
  <div class="solar-eclipse__occluder" style="--eclipse-occluder: #ffffff;"></div>
</div>

<!-- Note: Be sure to set --eclipse-occluder to match your container background color -->
