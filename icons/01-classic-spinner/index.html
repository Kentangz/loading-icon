Quick usage

<!-- 1. Import the stylesheet -->
<link rel="stylesheet" href="spinner.css" />

<!-- 2. Place the element -->
<div class="spinner"></div>

<!-- 3. Apply size + color modifiers -->
<div class="spinner spinner--lg spinner--success"></div>

<!-- 4. Override any token inline -->
<div class="spinner" style="--spinner-color:#e05c97; --spinner-size:44px;"></div>
