08 · Circular Progress

Quick usage

<link rel="stylesheet" href="circular-progress.css" />

<!-- Indeterminate (spinner) -->
<div class="circ-progress circ-progress--md circ-progress--primary">
  <svg class="circ-progress__svg" viewBox="0 0 44 44">
    <circle class="circ-progress__track" cx="22" cy="22" r="18"/>
    <circle class="circ-progress__fill"  cx="22" cy="22" r="18" pathLength="100"/>
  </svg>
</div>

<!-- Determinate — set --progress-value (0–100) -->
<div class="circ-progress circ-progress--determinate" style="--progress-value:75;">
  <svg class="circ-progress__svg" viewBox="0 0 44 44">
    <circle class="circ-progress__track" cx="22" cy="22" r="18"/>
    <circle class="circ-progress__fill"  cx="22" cy="22" r="18" pathLength="100"/>
  </svg>
</div>
