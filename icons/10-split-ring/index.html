Quick usage

<link rel="stylesheet" href="split-ring.css" />

<!-- Basic -->
<div class="split-ring"></div>

<!-- With modifiers -->
<div class="split-ring split-ring--lg split-ring--primary"></div>

<!-- Counter-clockwise -->
<div class="split-ring split-ring--ccw split-ring--danger"></div>

<!-- Two-tone arcs -->
<div class="split-ring split-ring--duo split-ring--primary" style="--ring-color-2:#E24B4A;"></div>

<!-- Custom via CSS properties -->
<div class="split-ring" style="--ring-color:#9333ea; --ring-speed:0.4s;"></div>
